. In the VM (Ubuntu)
+
[source,shell]
----
# Update
sudo apt update

# Install SSH
sudo apt install -y openssh-server

# Generate Keys
ssh-keygen -t rsa -b 4096 -q -N "" -f ~/.ssh/id_rsa

# Write down the IP
ip addr
----



. In the Host
+
[source,shell]
----
scp .ssh/id_rsa.pub osboxes@192.168.1.39:.ssh/authorized_keys
----
